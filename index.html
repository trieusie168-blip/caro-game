// ===== AI (ĐÃ TỐI ƯU HÓA ĐỂ TẤN CÔNG) ===== function aiMove(){ let moves = []; // Quét toàn bộ bảng để tìm các ô trống for(let i = 0; i < SIZE; i++){ for(let j = 0; j < SIZE; j++){ if(board[i][j] === ""){ // Chỉ xét các ô có hàng xóm trong phạm vi 2 ô để tăng tốc độ xử lý if (hasNeighbor(i, j)) { moves.push({ r: i, c: j, score: evaluate(i, j) }); } } } } if(moves.length === 0) return; // 1. KIỂM TRA THẮNG NGAY: Nếu có nước thắng, phải đi ngay for(let m of moves){ board[m.r][m.c] = "O"; if(checkWin(m.r, m.c, "O")){ board[m.r][m.c] = ""; return placeAI(m); } board[m.r][m.c] = ""; } // 2. KIỂM TRA CHẶN NGAY: Nếu đối thủ sắp thắng, phải chặn for(let m of moves){ board[m.r][m.c] = "X"; if(checkWin(m.r, m.c, "X")){ board[m.r][m.c] = ""; return placeAI(m); } board[m.r][m.c] = ""; } // 3. SẮP XẾP THEO ĐIỂM moves.sort((a, b) => b.score - a.score); // 4. CHỌN THEO LEVEL if(difficulty === "easy"){ let randomIdx = Math.floor(Math.random() * Math.min(moves.length, 10)); return placeAI(moves[randomIdx]); } if(difficulty === "normal"){ let top = moves.slice(0, 3); return placeAI(top[Math.floor(Math.random() * top.length)]); } return placeAI(moves[0]); // HARD: Luôn chọn nước đi điểm cao nhất } // Hàm bổ trợ để AI không quét các ô quá xa (tăng hiệu suất) function hasNeighbor(r, c) { for (let i = -2; i <= 2; i++) { for (let j = -2; j <= 2; j++) { if (board[r + i]?.[c + j] && board[r + i][c + j] !== "") return true; } } return false; } // ===== ĐIỀU CHỈNH TRỌNG SỐ TẤN CÔNG ===== function evaluate(r, c){ // Tăng vọt trọng số của "O" (AI) lên cao hơn "X" (Người chơi) // Giờ đây AI sẽ ưu tiên tự tạo thế cờ của mình hơn là đi theo đuôi bạn return score(r, c, "O") * 5 + score(r, c, "X") * 2; } function score(r, c, p){ let dirs = [[0, 1], [1, 0], [1, 1], [1, -1]]; let total = 0; for(let [dr, dc] of dirs){ let count = 1, open = 0; for(let i = 1; i < 5; i++){ let nr = r + dr * i, nc = c + dc * i; if(board[nr]?.[nc] == p) count++; else { if(board[nr]?.[nc] == "") open++; break; } } for(let i = 1; i < 5; i++){ let nr = r - dr * i, nc = c - dc * i; if(board[nr]?.[nc] == p) count++; else { if(board[nr]?.[nc] == "") open++; break; } } // Hệ thống điểm mới: Ưu tiên cực cao cho các thế cờ mở if(count >= 5) total += 10000000; // Thắng else if(count == 4 && open == 2) total += 5000000; // 4 con 2 đầu trống (vô địch) else if(count == 4 && open == 1) total += 1000000; // 4 con 1 đầu trống else if(count == 3 && open == 2) total += 500000; // 3 con 2 đầu trống else if(count == 3 && open == 1) total += 100000; else if(count == 2 && open == 2) total += 10000; else if(count == 2 && open == 1) total += 1000; } return total; }
